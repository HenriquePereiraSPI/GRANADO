    { month: 'Jan', sales: 120 }, { month: 'Fev', sales: 180 }, { month: 'Mar', sales: 150 },
    { month: 'Abr', sales: 220 }, { month: 'Mai', sales: 200 }, { month: 'Jun', sales: 260 }
  ];
</script>

bar-click (log no console) + props de yField (color · formatter · tooltipFormatter · borderRadius · opacity · barWidth · sort)

<granado-chart-bar id="c" type="grouped" title="Clique numa barra" show-values="true"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'month';
  c.yFields = [
    {
      field: 'sales', label: 'Vendas',
      color: '#1F7A3D',                                  // cor da série
      borderRadius: 6,                                   // pontas arredondadas
      barWidth: 26,                                      // largura fixa da barra
      sort: 1,                                           // ordem de render
      formatter: function (v) { return 'R$ ' + v; },     // rótulo/valor
      tooltipFormatter: function (ctx) {                 // HTML do tooltip
        return '<b>' + ctx.category + '</b><br>Vendas: R$ ' + ctx.value;
      }
    },
    {
      field: 'target', label: 'Meta',
      color: '#9A7520', opacity: 0.55, borderRadius: 6, sort: 2,
      formatter: function (v) { return 'R$ ' + v; }
    }
  ];
  c.data = [
    { month: 'Jan', sales: 120, target: 150 }, { month: 'Fev', sales: 180, target: 160 },
    { month: 'Mar', sales: 150, target: 170 }, { month: 'Abr', sales: 220, target: 190 }
  ];

  // clique numa barra -> loga os valores no console
  c.addEventListener('bar-click', function (ev) {
    console.log('[bar-click]', ev.detail);
  });
</script>

cores por item + slice-click (log no console) + formatter/tooltipFormatter

<granado-chart-pie id="c" type="donut" title="Clique numa fatia" show-values="true"></granado-chart-pie>
<script>
  var c = document.getElementById('c');
  c.xField = 'status';
  c.valueField = 'qtd';
  c.data = [
    { status: 'OK',       qtd: 320, color: '#1F7A3D' },
    { status: 'Alerta',   qtd: 60,  color: '#9A7520' },
    { status: 'Fora',     qtd: 40,  color: '#B5342F' }
  ];
  c.formatter = function (v) { return v + ' un'; };
  c.tooltipFormatter = function (ctx) {
    return '<b>' + ctx.label + '</b><br>' + ctx.value + ' un · ' + ctx.percent.toFixed(0) + '%';
  };

  // clique numa fatia -> loga no console + destaca (toggle)
  c.addEventListener('slice-click', function (ev) {
    console.log('[slice-click]', ev.detail);
  });
</script>
