Gráficos Granado

Exemplos visuais do <granado-chart-bar> — gráfico à esquerda, snippet que o gerou à direita.

type="simple" · uma série · valores + tooltip

<granado-chart-bar id="c" title="Vendas por mês" show-values="true"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'month';
  c.yFields = [{ field: 'sales', label: 'Vendas' }];
  c.data = [
    { month: 'Jan', sales: 120 }, { month: 'Fev', sales: 180 }, { month: 'Mar', sales: 150 },
    { month: 'Abr', sales: 220 }, { month: 'Mai', sales: 200 }, { month: 'Jun', sales: 260 }
  ];
</script>

type="grouped" · múltiplas séries (yFields) · show-export

<granado-chart-bar id="c" type="grouped" title="Vendas x Meta por mês" show-export="true"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'month';
  c.yFields = [{ field: 'sales', label: 'Vendas' }, { field: 'target', label: 'Meta' }];
  c.data = [
    { month: 'Jan', sales: 120, target: 150 }, { month: 'Fev', sales: 180, target: 160 },
    { month: 'Mar', sales: 150, target: 170 }, { month: 'Abr', sales: 220, target: 190 }
  ];

  // export por código (ex.: botão próprio):
  // c.exportPNG('vendas_meta');  c.exportSVG('vendas_meta');
</script>

type="target" · faixa min→max (horizontal)

<granado-chart-bar id="c" type="target" title="Faixa salarial por cargo"
    orientation="horizontal" show-values="true"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'employee';
  c.yFields = [{ field: 'min' }, { field: 'max', label: 'Faixa (R$)' }]; // [low, high]
  c.data = [
    { employee: 'João', min: 4000, max: 7000 },
    { employee: 'Maria', min: 5000, max: 8500 },
    { employee: 'Ana', min: 3500, max: 6200 }
  ];
</script>

x2-field="region" · segundo agrupador (formato longo)

<granado-chart-bar id="c" type="grouped" title="Vendas por trimestre e região"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'quarter';
  c.x2Field = 'region';              // 2º agrupador: uma série por região
  c.yFields = [{ field: 'sales' }];  // valor usado no pivot
  c.data = [
    { quarter: 'Q1', region: 'Sul', sales: 120 }, { quarter: 'Q1', region: 'Norte', sales: 90 },
    { quarter: 'Q2', region: 'Sul', sales: 150 }, { quarter: 'Q2', region: 'Norte', sales: 110 },
    { quarter: 'Q3', region: 'Sul', sales: 170 }, { quarter: 'Q3', region: 'Norte', sales: 140 }
  ];
</script>

cores/opacidade/borderRadius por série

<granado-chart-bar id="c" type="grouped" title="Estilo customizado"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'month';
  c.yFields = [
    { field: 'sales',  label: 'Vendas', color: '#1F7A3D', borderRadius: 6 },
    { field: 'target', label: 'Meta',   color: '#9A7520', opacity: 0.55, borderRadius: 6 }
  ];
  c.data = [
    { month: 'Jan', sales: 120, target: 150 }, { month: 'Fev', sales: 180, target: 160 },
    { month: 'Mar', sales: 150, target: 170 }, { month: 'Abr', sales: 220, target: 190 }
  ];
</script>

barRole · vendas (barra) + meta ("line") + limite ("dashed")

<granado-chart-bar id="c" type="grouped" title="Vendas x Meta x Limite"></granado-chart-bar>
<script>
  var c = document.getElementById('c');
  c.xField = 'month';
  c.yFields = [
    { field: 'sales',  label: 'Vendas' },                          // barRole 'default'
    { field: 'target', label: 'Meta',   barRole: 'line',   color: '#8C1A1A' },
    { field: 'limite', label: 'Limite', barRole: 'dashed', color: '#1A4A8C' }
  ];
  c.data = [
    { month: 'Jan', sales: 120, target: 150, limite: 180 },
    { month: 'Fev', sales: 180, target: 160, limite: 190 },
    { month: 'Mar', sales: 150, target: 170, limite: 200 },
    { month: 'Abr', sales: 220, target: 190, limite: 210 }
  ];
</script>

<granado-chart-bar id="c" title="Vendas por mês (meta 200)" show-values="true"></granado-chart-bar>

  c.xField = 'month';
  c.yFields = [{ field: 'sales', label: 'Vendas' }];
  c.overallTarget = { value: 200, label: 'Meta 200' }; // ou só: c.overallTarget = 200

    { month: 'Abr', sales: 220 }, { month: 'Mai', sales: 200 }, { month: 'Jun', sales: 260 }
  ];
</script>
